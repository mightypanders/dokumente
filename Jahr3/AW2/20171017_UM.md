## Code Beispiel 
```csharp
public void Kuerze() {
    // größten gemeinsamen Teiler mit dem Euklidischen Algorithmus bestimmen
    if (zaehler != 0) {
    int ggt = 0;
    int az = Math.Abs(zaehler);
    int an = Math.Abs(nenner);
    do {
    if (az == an)
      ggt = az;
    else  
      if (az > an)  
       az = az ‐ an;
      else
       an = an ‐ az;
    } while (ggt == 0);
    
    zaehler /= ggt;
    nenner  /= ggt;
    }
}
```
